Score analysis in ExtractTopFeatures in CountClust

Kushal K Dey

8/24/2017

PBMC (sorted + unsorted) 68K data analysis

topic_clus <- get(load("../output/10X_genomics_pooled_sort_unsort_maptpx.rda"))

K=5

out <- ExtractTopFeatures(topic_clus[[5]]$theta, top_features = 100, method = "poisson", options = "max")
indices <- out$indices
scores <- out$scores
par(mfrow = c(3,3))
for(m in 1:5){
  plot(log(1:100), log(scores[m,]+1), col="black", pch=20, ylab = "scores", xlab="top 100 genes (sorted)",
       main = paste0("cluster-", m))
}

QQplot (K=8)

out <- ExtractTopFeatures(topic_clus[[8]]$theta, top_features = 100, method = "poisson", options = "max")
indices <- out$indices
scores <- out$scores

ll <- vector(mode="list", length=dim(scores)[1])
par(mfrow = c(3,3))
for(m in 1:dim(scores)[1]){
  out <- qqnorm(log(scores[m,]+1), col="red", pch=20, cex = 1, main = paste0("QQplot: cluster - ", m))
  ll[[m]] <- cbind(out$x, (out$y - mean(out$y))/(sd(out$y)))
  qqline(log(scores[m,]+1))
}

par(mfrow = c(3,3))
for(m in 1:dim(scores)[1]){
plot(density(abs(ll[[m]][,1] - ll[[m]][,2])), col = "red", main = "density of distance between quantiles", xlab = "distance - qq", ylab = "density")
}

par(mfrow = c(3,3))
for(m in 1:dim(scores)[1]){
plot(abs(ll[[m]][,1] - ll[[m]][,2]), col = "red", main = "abs diff between quantiles", xlab = "distance - qq", ylab = "density")
abline(0,0)
}
